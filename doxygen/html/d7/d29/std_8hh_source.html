27 /*
28  XDLType is the base class of all XDL-generated code.
29  Because any information in the class will increase the runtime size of
30  all children instead of storing a string typename inside XDLType, we will use
31  a table At constructor time, the built-in types look up the offset in the
32  table and store that (a 32-bit quantity) For a user-defined type, it
33  registers the name in a HashMap in SymbolTable, and then quickly turns the
34  string into an offset into the type table. This means the overhead of XDLType
35  is:
36 
37  1. a pointer to the VFT since there are virtual functions
38  2. the offset to the name in the array
39 
40  This is a total of 12 bytes, or 8 for a 32-bit platform.
41 
42  For speed, all methods that return a string object should return
43  const string& The string must already exist so that nothing is allocated
44  */

53 class XDLType {
54  protected:
55  const static std::string empty;
56  uint32_t nameOffset;
57  static DynArray<const XDLType*> types;
58  static DynArray<std::string> typeNames; // the list of all unique names in the system
59  static std::unordered_map<std::string, uint32_t> byName;
60  static void addType(const XDLType* type);
61 
62  public:
63  static void classInit();
64  static void classCleanup();
65  static uint32_t computeNameOffset(const std::string& typeName) {
66  uint32_t nameOffset;
67  if (byName.find(typeName) != byName.end())
68  nameOffset = byName.at(typeName);
69  else {
70  nameOffset = byName[typeName] = typeNames.size();
71  typeNames.add(typeName);
72  }
73  return nameOffset;
74  }
75  static uint32_t computeNameOffset(DataType t) {
76  uint32_t nameOffset = 0;
77  return nameOffset;
78  }
79  XDLType() {}
80  XDLType(const std::string& typeName)
81  : nameOffset(computeNameOffset(typeName)) {}
82  XDLType(DataType t) : nameOffset(computeNameOffset(t)) {}
83  virtual void write(Buffer& b) const = 0;
84  virtual void writeMeta(Buffer& buf) const;
85  virtual uint32_t size() const = 0;
86  static const XDLType* getBuiltinType(DataType dt) {
87  return types[uint32_t(dt)];
88  }
89 
90  virtual DataType getDataType() const = 0;
91  // TODO: virtual void generateCode() = 0;
92  const std::string& getTypeName() const { return typeNames[nameOffset]; }
93  // using this XDL type as ASCII text, not formatted
94  // each class must implement its own iterator?
95  virtual XDLIterator* createIterator();
96  virtual void display(Buffer& binaryIn, Buffer& asciiOut) const;
97  // TODO: for efficiency, add another display function since we usually do NOT
98  // want to know w,h virtual void display(Buffer& in, Canvas* c, const Style*
99  // s, float x0, float y0) const; using this XDL type, output formatted
100  void format(Buffer& binaryIn, Buffer& asciiOut, const char fmt[]) const {}
101  /* read from metadata and create a struct of all the variables specified
102  if any variables are structures, then this can get recursive
103  */
104  static void readMeta(XDLCompiler* compiler, Buffer& in, uint32_t count, Struct* s);
105  static const Struct* read(Buffer& in);
106  static DataType readType(Buffer& in);
107 };

631  void add(const T& e) { impl.push_back(e); }
632 #if 0
633  template <class... Args>
634  void add(Args&&... args) {
635  impl.emplace_back(args);
636  }

639  return impl.size(); // TODO: * T.size();

642  buf.write(uint16_t(impl.size()));
643  for (uint32_t i = 0; i < impl.size(); i++)
644  impl[i].write(buf);
645  }

647  buf.write(DataType::LIST16);
648  if (impl.size() == 0) return;
